Сборка
colcon build --packages-select verter
colcon build --symlink-install # Рекомендуемый способ сборки для разработки
source install/setup.bash

Запуск отдельных нод (для отладки)
ros2 run verter vosk_node

Публикация сообщения в топик (для отладки)
ros2 topic pub /verter_commands std_msgs/msg/String '{data: "go_forward"}' --once
ros2 topic pub /verter_commands std_msgs/msg/String '{data: "stop"}' --once

Запуск системы
# Запуск основной системы (движение, датчики)
ros2 launch verter verter_system.launch.py

# Запуск системы распознавания речи
ros2 launch verter speech_system.launch.py

ros2 launch verter main.launch.py

Полезные команды
# Просмотр активных портов
ls -l /dev/ttyA* /dev/ttyU*

# Проверка доступных аудиоустройств
python3 -c "import sounddevice as sd; print(sd.query_devices())"
